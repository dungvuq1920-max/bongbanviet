                    <div class="flex justify-between items-center">
                        <span class="text-[9px] font-bold text-red-600 bg-red-50 px-2 py-0.5 rounded uppercase tracking-widest">${p.subCategory}</span>
                        <span class="text-[9px] font-bold text-slate-400 uppercase tracking-widest">${p.brand}</span>
                    </div>
                    <h3 class="font-bold text-slate-800 text-sm mt-3 line-clamp-1 group-hover:text-red-600">${p.name}</h3>
                    <div class="flex justify-between items-center mt-3">
                        <span class="text-xs text-slate-400 font-medium italic">Chính hãng</span>
                        <span class="text-slate-900 font-black text-sm italic">${p.price}</span>
                    </div>
                </div>
            `).join('');

            document.querySelectorAll('.product-item').forEach(item => {
                item.addEventListener('mouseenter', () => { if (pinnedProductId === null) showPopups(item); });
                item.addEventListener('mouseleave', () => { if (pinnedProductId === null) hidePopups(); });
            });
        }

        function togglePin(element, event) {
            event.stopPropagation();
            const id = element.dataset.id;
            if (pinnedProductId == id) {
                pinnedProductId = null;
                element.classList.remove('pinned');
                hidePopups();
            } else {
                document.querySelectorAll('.product-item').forEach(i => i.classList.remove('pinned'));
                pinnedProductId = id;
                element.classList.add('pinned');
                showPopups(element);
            }
        }

        document.body.addEventListener('click', () => {
            if (pinnedProductId !== null) {
                pinnedProductId = null;
                document.querySelectorAll('.product-item').forEach(i => i.classList.remove('pinned'));
                hidePopups();
            }
        });

        function showPopups(element) {
            const p = products.find(prod => prod.id == element.dataset.id);
            const rect = element.getBoundingClientRect();

            document.getElementById('p-name').textContent = p.name;
            document.getElementById('p-price').textContent = p.price;
            document.getElementById('p-desc').textContent = p.desc;
            document.getElementById('p-cat-tag').textContent = `${p.brand} - ${p.subCategory}`;
            document.getElementById('p-specs').innerHTML = Object.entries(p.specs).map(([key, value]) => `
                <div>
                    <div class="flex justify-between text-[8px] font-bold text-slate-500 mb-1 uppercase">
                        <span>${key}</span>
                        <span class="text-red-600">${value}%</span>
                    </div>
                    <div class="h-1 bg-slate-100 rounded-full overflow-hidden"><div class="h-full bg-red-600" style="width: ${value}%"></div></div>
                </div>
            `).join('');

            const centerY = rect.top + rect.height / 2;
            let prX = rect.right + 50;
            let prY = centerY - 100;
            if (prX + 288 > window.innerWidth) prX = rect.left - 338;
            popupRight.style.left = `${prX}px`;
            popupRight.style.top = `${prY}px`;
            popupRight.classList.add('visible');

            const centerX = rect.left + rect.width / 2;
            let pbY = rect.bottom + 50;
            let pbX = centerX - 160;
            if (pbY + 200 > window.innerHeight) pbY = rect.top - 250;
            popupBottom.style.left = `${pbX}px`;
            popupBottom.style.top = `${pbY}px`;
            popupBottom.classList.add('visible');

            const startX_R = (prX > rect.right) ? rect.right : rect.left;
            const endX_R = (prX > rect.right) ? prX - 5 : prX + 288 + 5;
            drawLine(lineRight, startX_R, centerY, endX_R, centerY);

            const startY_B = (pbY > rect.bottom) ? rect.bottom : rect.top;
            const endY_B = (pbY > rect.bottom) ? pbY - 5 : pbY + 200 + 5;
            drawLine(lineBottom, centerX, startY_B, centerX, endY_B);
        }

        function drawLine(path, x1, y1, x2, y2) {
            path.setAttribute('d', `M ${x1} ${y1} L ${x2} ${y2}`);
            path.classList.remove('active-line');
            void path.offsetWidth;
            path.classList.add('active-line');
        }

        function hidePopups() {
            popupRight.classList.remove('visible');
            popupBottom.classList.remove('visible');
            lineRight.classList.remove('active-line');
            lineBottom.classList.remove('active-line');
        }

        function renderSubMenu(cat) {
            if (!subMenuData[cat]) {
                subMenuContainer.classList.remove('show');
                return;
            }
            subMenuContainer.innerHTML = subMenuData[cat].map(s => `
                <button class="sub-tab px-5 py-2 border-2 border-slate-200 text-slate-500 rounded-xl font-bold text-[10px] uppercase tracking-widest transition-all whitespace-nowrap ${s === currentSubCategory ? 'active bg-red-600 !text-white !border-red-600 shadow-lg shadow-red-200' : 'bg-white hover:border-red-400'}" data-sub="${s}">${s}</button>
            `).join('');
            subMenuContainer.classList.add('show');
            document.querySelectorAll('.sub-tab').forEach(tab => {
                tab.addEventListener('click', (e) => {
                    e.stopPropagation();
                    currentSubCategory = tab.dataset.sub;
                    renderSubMenu(currentCategory);
                    pinnedProductId = null;
                    hidePopups();
                    renderProducts();
                });
            });
        }

        document.querySelectorAll('.nav-item').forEach(tab => {
            tab.addEventListener('click', (e) => {
                e.stopPropagation();
                document.querySelectorAll('.nav-item').forEach(t => t.classList.remove('active', 'text-slate-900'));
                document.querySelectorAll('.nav-item').forEach(t => t.classList.add('text-slate-500'));
                tab.classList.add('active', 'text-slate-900');
                tab.classList.remove('text-slate-500');
                currentCategory = tab.dataset.category;
                currentSubCategory = 'Tất cả';
                catTitle.textContent = `${tab.textContent} chuyên nghiệp`;
                pinnedProductId = null;
                hidePopups();
                renderSubMenu(currentCategory);
                renderProducts();
            });
        });

        [filterSearch, filterBrand, filterPrice].forEach(el => {
            el.addEventListener('input', renderProducts);
            el.addEventListener('change', renderProducts);
        });

        filterReset.addEventListener('click', () => {
            filterSearch.value = '';
            filterBrand.value = 'all';
            filterPrice.value = 'all';
            renderProducts();
        });

        renderSubMenu(currentCategory);
        renderProducts();
    </script>
</body>
